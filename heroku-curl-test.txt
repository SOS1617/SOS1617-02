
////////////////////////////////////////////CURL A HEROKU SMI-STATS///////////////////////////////////////////////////////

//Get a colección
curl -XGET -H 'Content-type: application/json'  'http://sos1617-02.herokuapp.com/api/v1/smi_stats'

//Post a colección
curl -XPOST -H 'Content-type: application/json' -d '{ "country": "France", "year": "2017", "smi-year": "1480.3", "smi-year-variation": "0.93" }' 'http://sos1617-02.herokuapp.com/api/v1/smi_stats'

//Get a elemento
curl -XGET -H 'Content-type: application/json'  'http://sos1617-02.herokuapp.com/api/v1/smi_stats/China'

//PUT a colección
curl -XPUT -H 'Content-type: application/json' -d '{ "country": "Spain", "year": "2017", "smi-year": "1234.8", "smi-year-variation": "69.9" }' 'http://sos1617-02.herokuapp.com/api/v1/smi_stats'

//PUT a elemento
curl -XPUT -H "Content-type: application/json" -d '{ "country": "Spain", "year": "2017", "smi-year": "1234.8", "smi-year-variation": "69.9" }' 'http://sos1617-02.herokuapp.com/api/v1/smi_stats/Spain'

//GET a elemento
curl -XGET -H "Content-type: application/json"  'http://sos1617-02.herokuapp.com/api/v1/smi_stats/Spain'

//DELETE a elemento
curl -XDELETE -H "Content-type: application/json"  'http://sos1617-02.herokuapp.com/api/v1/smi_stats/Spain'

//DELETE a colección
curl -XDELETE -H "Content-type: application/json"  'http://sos1617-02.herokuapp.com/api/v1/smi_stats'
